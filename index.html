Configurações

Conecte uma planilha Google compartilhada para que dois aparelhos vejam os mesmos gastos.

1Crie a planilha

  1. Acesse sheets.new e crie uma planilha em branco.
  2. Compartilhe com a outra pessoa (botão Compartilhar).

2Adicione o script

  1. Na planilha, clique em Extensões → Apps Script.
  2. Apague o conteúdo e cole o código abaixo.
  3. Salve (💾) e clique em Implantar → Nova implantação.
  4. Tipo: App da Web. Acesso: Qualquer pessoa. Clique em Implantar.
  5. Copie a URL gerada e cole no campo abaixo.
function doPost(e) { try { const data = JSON.parse(e.postData.contents); const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); ensureHeader(sheet); const action = data.action || 'add'; if (action === 'add') return addRow(sheet, data); if (action === 'requestDelete') return setStatus(sheet, data.id, 'pendente', data.solicitadoPor || ''); if (action === 'cancelDelete') return setStatus(sheet, data.id, '', ''); if (action === 'rejectDelete') return setStatus(sheet, data.id, '', ''); if (action === 'approveDelete') return deleteRow(sheet, data.id); return jsonErr('Ação desconhecida: ' + action); } catch (err) { return jsonErr(err); } } function ensureHeader(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','Registrado','DataGasto','Local','Item','Valor','Tipo','Categoria','StatusRemocao','SolicitadoPor']); return; } const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; ['DataGasto','Categoria','StatusRemocao','SolicitadoPor'].forEach(col => { if (head.indexOf(col) === -1) { sheet.getRange(1, sheet.getLastColumn() + 1).setValue(col); } }); } function addRow(sheet, data) { const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const v = { 'ID': data.id, 'Registrado': new Date(data.data), 'Data': new Date(data.data), 'DataGasto': data.dataGasto ? new Date(data.dataGasto + 'T12:00:00') : new Date(data.data), 'Local': data.local, 'Item': data.item, 'Valor': Number(data.valor), 'Tipo': data.tipo, 'Categoria': data.categoria || '', 'StatusRemocao': '', 'SolicitadoPor': '' }; sheet.appendRow(head.map(h => v[h] !== undefined ? v[h] : '')); return jsonOk({}); } function findRow(sheet, id) { const vals = sheet.getDataRange().getValues(); for (let i = 1; i < vals.length; i++) { if (String(vals[i][0]) === String(id)) return i + 1; } return -1; } function setStatus(sheet, id, status, por) { const row = findRow(sheet, id); if (row < 0) return jsonErr('ID não encontrado'); const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const cS = head.indexOf('StatusRemocao') + 1; const cP = head.indexOf('SolicitadoPor') + 1; if (cS > 0) sheet.getRange(row, cS).setValue(status); if (cP > 0) sheet.getRange(row, cP).setValue(por); return jsonOk({}); } function deleteRow(sheet, id) { const row = findRow(sheet, id); if (row < 0) return jsonErr('ID não encontrado'); sheet.deleteRow(row); return jsonOk({}); } function jsonOk(extra) { return ContentService.createTextOutput(JSON.stringify(Object.assign({ok:true}, extra))) .setMimeType(ContentService.MimeType.JSON); } function jsonErr(err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:String(err)})) .setMimeType(ContentService.MimeType.JSON); } function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const vals = sheet.getDataRange().getValues(); if (vals.length === 0) return jsonOk({data:[]}); const head = vals[0]; const idx = (n) => head.indexOf(n); const rows = vals.slice(1).map(r => ({ id: r[idx('ID')], data: r[idx('Registrado')] || r[idx('Data')], dataGasto: idx('DataGasto') >= 0 ? r[idx('DataGasto')] : null, local: r[idx('Local')], item: r[idx('Item')], valor: r[idx('Valor')], tipo: r[idx('Tipo')], categoria: idx('Categoria') >= 0 ? (r[idx('Categoria')] || '') : '', statusRemocao: idx('StatusRemocao') >= 0 ? (r[idx('StatusRemocao')] || '') : '', solicitadoPor: idx('SolicitadoPor') >= 0 ? (r[idx('SolicitadoPor')] || '') : '' })).filter(r => r.id); return jsonOk({data: rows}); }
