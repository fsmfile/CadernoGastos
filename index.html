Configurações

Conecte uma planilha Google compartilhada para que dois aparelhos vejam os mesmos gastos.

1Crie a planilha

  1. Acesse sheets.new e crie uma planilha em branco.
  2. Compartilhe com a outra pessoa (botão Compartilhar).

2Adicione o script

  1. Na planilha, clique em Extensões → Apps Script.
  2. Apague o conteúdo e cole o código abaixo.
  3. Salve (💾) e clique em Implantar → Nova implantação.
  4. Tipo: App da Web. Acesso: Qualquer pessoa. Clique em Implantar.
  5. Copie a URL gerada e cole no campo abaixo.
function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = data.action || 'add'; if (action === 'add') return addGasto(ss, data); if (action === 'requestDelete') return setStatus(ss, data.id, 'pendente', data.solicitadoPor || ''); if (action === 'cancelDelete') return setStatus(ss, data.id, '', ''); if (action === 'rejectDelete') return setStatus(ss, data.id, '', ''); if (action === 'approveDelete') return deleteGasto(ss, data.id); if (action === 'addTrip') return addViagem(ss, data); if (action === 'updateTrip') return updateViagem(ss, data); if (action === 'deleteTrip') return deleteViagem(ss, data.id); return jsonErr('Ação desconhecida: ' + action); } catch (err) { return jsonErr(err); } } function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const gSheet = getGastosSheet(ss); const vSheet = getViagensSheet(ss); ensureGastosHeader(gSheet); ensureViagensHeader(vSheet); const gv = gSheet.getDataRange().getValues(); const gh = gv[0] || []; const gi = (n) => gh.indexOf(n); const gastos = gv.slice(1).map(r => ({ id: r[gi('ID')], viagemId: gi('ViagemId') >= 0 ? r[gi('ViagemId')] : '', data: r[gi('Registrado')] || r[gi('Data')], dataGasto: gi('DataGasto') >= 0 ? r[gi('DataGasto')] : null, local: r[gi('Local')], item: r[gi('Item')], valor: r[gi('Valor')], tipo: r[gi('Tipo')], categoria: gi('Categoria') >= 0 ? (r[gi('Categoria')] || '') : '', statusRemocao: gi('StatusRemocao') >= 0 ? (r[gi('StatusRemocao')] || '') : '', solicitadoPor: gi('SolicitadoPor') >= 0 ? (r[gi('SolicitadoPor')] || '') : '' })).filter(r => r.id); const vv = vSheet.getDataRange().getValues(); const vh = vv[0] || []; const vi = (n) => vh.indexOf(n); const viagens = vv.slice(1).map(r => ({ id: r[vi('ID')], nome: r[vi('Nome')], inicio: vi('Inicio') >= 0 ? r[vi('Inicio')] : null, fim: vi('Fim') >= 0 ? r[vi('Fim')] : null, orcamentoDebito: Number(r[vi('OrcamentoDebito')]) || 0, orcamentoCredito: Number(r[vi('OrcamentoCredito')]) || 0, criadaEm: vi('CriadaEm') >= 0 ? r[vi('CriadaEm')] : null, arquivada: vi('Arquivada') >= 0 ? !!r[vi('Arquivada')] : false })).filter(r => r.id); return jsonOk({ data: gastos, viagens: viagens }); } function getGastosSheet(ss) { let s = ss.getSheetByName('Gastos'); if (!s) { s = ss.getSheets()[0]; if (s.getName() !== 'Gastos' && s.getName() !== 'Viagens') s.setName('Gastos'); else if (s.getName() === 'Viagens') s = ss.insertSheet('Gastos', 0); } return s; } function getViagensSheet(ss) { let s = ss.getSheetByName('Viagens'); if (!s) s = ss.insertSheet('Viagens'); return s; } function ensureGastosHeader(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','ViagemId','Registrado','DataGasto','Local','Item','Valor','Tipo','Categoria','StatusRemocao','SolicitadoPor']); return; } const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; ['ViagemId','DataGasto','Categoria','StatusRemocao','SolicitadoPor'].forEach(col => { if (head.indexOf(col) === -1) { sheet.getRange(1, sheet.getLastColumn() + 1).setValue(col); } }); } function ensureViagensHeader(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','Nome','Inicio','Fim','OrcamentoDebito','OrcamentoCredito','CriadaEm','Arquivada']); } } function addGasto(ss, data) { const sheet = getGastosSheet(ss); ensureGastosHeader(sheet); const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const v = { 'ID': data.id, 'ViagemId': data.viagemId || '', 'Registrado': new Date(data.data), 'Data': new Date(data.data), 'DataGasto': data.dataGasto ? new Date(data.dataGasto + 'T12:00:00') : new Date(data.data), 'Local': data.local, 'Item': data.item, 'Valor': Number(data.valor), 'Tipo': data.tipo, 'Categoria': data.categoria || '', 'StatusRemocao': '', 'SolicitadoPor': '' }; sheet.appendRow(head.map(h => v[h] !== undefined ? v[h] : '')); return jsonOk({}); } function findRow(sheet, id, col) { const vals = sheet.getDataRange().getValues(); const cIdx = col ? vals[0].indexOf(col) : 0; for (let i = 1; i < vals.length; i++) { if (String(vals[i][cIdx]) === String(id)) return i + 1; } return -1; } function setStatus(ss, id, status, por) { const sheet = getGastosSheet(ss); ensureGastosHeader(sheet); const row = findRow(sheet, id, 'ID'); if (row < 0) return jsonErr('ID não encontrado'); const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const cS = head.indexOf('StatusRemocao') + 1; const cP = head.indexOf('SolicitadoPor') + 1; if (cS > 0) sheet.getRange(row, cS).setValue(status); if (cP > 0) sheet.getRange(row, cP).setValue(por); return jsonOk({}); } function deleteGasto(ss, id) { const sheet = getGastosSheet(ss); const row = findRow(sheet, id, 'ID'); if (row < 0) return jsonErr('ID não encontrado'); sheet.deleteRow(row); return jsonOk({}); } function addViagem(ss, data) { const sheet = getViagensSheet(ss); ensureViagensHeader(sheet); sheet.appendRow([ data.id, data.nome, data.inicio ? new Date(data.inicio + 'T12:00:00') : '', data.fim ? new Date(data.fim + 'T12:00:00') : '', Number(data.orcamentoDebito) || 0, Number(data.orcamentoCredito) || 0, new Date(), false ]); return jsonOk({}); } function updateViagem(ss, data) { const sheet = getViagensSheet(ss); ensureViagensHeader(sheet); const row = findRow(sheet, data.id, 'ID'); if (row < 0) return jsonErr('Viagem não encontrada'); const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const map = { 'Nome': data.nome, 'Inicio': data.inicio ? new Date(data.inicio + 'T12:00:00') : '', 'Fim': data.fim ? new Date(data.fim + 'T12:00:00') : '', 'OrcamentoDebito': Number(data.orcamentoDebito) || 0, 'OrcamentoCredito': Number(data.orcamentoCredito) || 0, 'Arquivada': !!data.arquivada }; Object.keys(map).forEach(k => { if (map[k] === undefined) return; const col = head.indexOf(k) + 1; if (col > 0) sheet.getRange(row, col).setValue(map[k]); }); return jsonOk({}); } function deleteViagem(ss, id) { const sheet = getViagensSheet(ss); const row = findRow(sheet, id, 'ID'); if (row < 0) return jsonErr('Viagem não encontrada'); sheet.deleteRow(row); return jsonOk({}); } function jsonOk(extra) { return ContentService.createTextOutput(JSON.stringify(Object.assign({ok:true}, extra))) .setMimeType(ContentService.MimeType.JSON); } function jsonErr(err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:String(err)})) .setMimeType(ContentService.MimeType.JSON); }
