Configurações

Conecte uma planilha Google compartilhada para que dois aparelhos vejam os mesmos gastos.

1Crie a planilha

  1. Acesse sheets.new e crie uma planilha em branco.
  2. Compartilhe com a outra pessoa (botão Compartilhar).

2Adicione o script

  1. Na planilha, clique em Extensões → Apps Script.
  2. Apague o conteúdo e cole o código abaixo.
  3. Salve (💾) e clique em Implantar → Nova implantação.
  4. Tipo: App da Web. Acesso: Qualquer pessoa. Clique em Implantar.
  5. Copie a URL gerada e cole no campo abaixo.
function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const action = data.action || 'add'; if (action === 'add') return addGasto(ss, data); if (action === 'requestDelete') return setStatus(ss, data.id, 'pendente', data.solicitadoPor || ''); if (action === 'cancelDelete') return setStatus(ss, data.id, '', ''); if (action === 'rejectDelete') return setStatus(ss, data.id, '', ''); if (action === 'approveDelete') return deleteGasto(ss, data.id); if (action === 'addTrip') return addViagem(ss, data); if (action === 'updateTrip') return updateViagem(ss, data); if (action === 'deleteTrip') return deleteViagem(ss, data.id); if (action === 'adjustBalance') return addAjuste(ss, data); if (action === 'assignTrip') return assignTripToGastos(ss, data); if (action === 'ping') return jsonOk({pong: true}); return jsonErr('Ação desconhecida: ' + action); } catch (err) { return jsonErr(err); } } function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const gSheet = getGastosSheet(ss); const vSheet = getViagensSheet(ss); const sSheet = getSaldosSheet(ss); ensureGastosHeader(gSheet); ensureViagensHeader(vSheet); ensureSaldosHeader(sSheet); const gv = gSheet.getDataRange().getValues(); const gh = gv[0] || []; const gi = (n) => gh.indexOf(n); const gastos = gv.slice(1).map(r => ({ id: r[gi('ID')], viagemId: gi('ViagemId') >= 0 ? r[gi('ViagemId')] : '', scope: gi('Scope') >= 0 ? (r[gi('Scope')] || '') : '', data: r[gi('Registrado')] || r[gi('Data')], dataGasto: gi('DataGasto') >= 0 ? r[gi('DataGasto')] : null, local: r[gi('Local')], item: r[gi('Item')], valor: r[gi('Valor')], tipo: r[gi('Tipo')], categoria: gi('Categoria') >= 0 ? (r[gi('Categoria')] || '') : '', statusRemocao: gi('StatusRemocao') >= 0 ? (r[gi('StatusRemocao')] || '') : '', solicitadoPor: gi('SolicitadoPor') >= 0 ? (r[gi('SolicitadoPor')] || '') : '' })).filter(r => r.id); const vv = vSheet.getDataRange().getValues(); const vh = vv[0] || []; const vi = (n) => vh.indexOf(n); const viagens = vv.slice(1).map(r => ({ id: r[vi('ID')], nome: r[vi('Nome')], inicio: vi('Inicio') >= 0 ? r[vi('Inicio')] : null, fim: vi('Fim') >= 0 ? r[vi('Fim')] : null, orcamentoDebito: Number(r[vi('OrcamentoDebito')]) || 0, orcamentoCredito: Number(r[vi('OrcamentoCredito')]) || 0, criadaEm: vi('CriadaEm') >= 0 ? r[vi('CriadaEm')] : null, arquivada: vi('Arquivada') >= 0 ? !!r[vi('Arquivada')] : false })).filter(r => r.id); const sv = sSheet.getDataRange().getValues(); const sh = sv[0] || []; const si = (n) => sh.indexOf(n); const ajustes = sv.slice(1).map(r => ({ id: r[si('ID')], data: r[si('Data')], tipo: r[si('Tipo')], valorAntes: Number(r[si('ValorAntes')]) || 0, valorDepois: Number(r[si('ValorDepois')]) || 0, motivo: r[si('Motivo')] || '', por: r[si('Por')] || '' })).filter(r => r.id); return jsonOk({ data: gastos, viagens: viagens, ajustes: ajustes }); } function getGastosSheet(ss) { let s = ss.getSheetByName('Gastos'); if (!s) { s = ss.getSheets()[0]; if (s.getName() !== 'Gastos' && s.getName() !== 'Viagens' && s.getName() !== 'Saldos') s.setName('Gastos'); else if (s.getName() !== 'Gastos') s = ss.insertSheet('Gastos', 0); } return s; } function getViagensSheet(ss) { let s = ss.getSheetByName('Viagens'); if (!s) s = ss.insertSheet('Viagens'); return s; } function getSaldosSheet(ss) { let s = ss.getSheetByName('Saldos'); if (!s) s = ss.insertSheet('Saldos'); return s; } function ensureGastosHeader(sheet) { // Esquema completo desejado (ordem importa só para planilhas novas) var SCHEMA = ['ID','ViagemId','Scope','Registrado','DataGasto','Local','Item','Valor','Tipo','Categoria','StatusRemocao','SolicitadoPor']; if (sheet.getLastRow() === 0) { sheet.appendRow(SCHEMA); return; } var lastCol = sheet.getLastColumn(); var head = sheet.getRange(1, 1, 1, lastCol).getValues()[0]; // Adiciona apenas colunas que AINDA NÃO existem // (indexOf ignora as duplicadas, mas evita adicionar uma terceira) SCHEMA.forEach(function(col) { if (head.indexOf(col) === -1) { sheet.getRange(1, sheet.getLastColumn() + 1).setValue(col); head.push(col); } }); } function ensureViagensHeader(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','Nome','Inicio','Fim','OrcamentoDebito','OrcamentoCredito','CriadaEm','Arquivada']); } } function ensureSaldosHeader(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','Data','Tipo','ValorAntes','ValorDepois','Motivo','Por']); } } function addAjuste(ss, data) { const sheet = getSaldosSheet(ss); ensureSaldosHeader(sheet); sheet.appendRow([ data.id, new Date(data.data), data.tipo, Number(data.valorAntes) || 0, Number(data.valorDepois) || 0, data.motivo || '', data.por || '' ]); return jsonOk({}); } // Atribui em massa: para cada id em data.ids, define ViagemId e Scope. // Usado pela migração de gastos órfãos e pelo botão "Reparar dados". function assignTripToGastos(ss, data) { const sheet = getGastosSheet(ss); ensureGastosHeader(sheet); const ids = Array.isArray(data.ids) ? data.ids.map(String) : []; if (ids.length === 0) return jsonOk({ updated: 0 }); const viagemId = String(data.viagemId || ''); const scope = String(data.scope || 'viagem'); const vals = sheet.getDataRange().getValues(); const head = vals[0]; const cId = head.indexOf('ID'); const cV = head.indexOf('ViagemId') + 1; const cS = head.indexOf('Scope') + 1; if (cId < 0 || cV === 0) return jsonErr('Colunas ID/ViagemId não encontradas'); const idSet = new Set(ids); let updated = 0; for (let i = 1; i < vals.length; i++) { if (idSet.has(String(vals[i][cId]))) { sheet.getRange(i + 1, cV).setValue(viagemId); if (cS > 0) sheet.getRange(i + 1, cS).setValue(scope); updated++; } } return jsonOk({ updated: updated }); } function addGasto(ss, data) { const sheet = getGastosSheet(ss); ensureGastosHeader(sheet); const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const v = { 'ID': data.id, 'ViagemId': data.viagemId || '', 'Scope': data.scope || 'viagem', 'Registrado': new Date(data.data), 'Data': new Date(data.data), 'DataGasto': data.dataGasto ? new Date(data.dataGasto + 'T12:00:00') : new Date(data.data), 'Local': data.local, 'Item': data.item, 'Valor': Number(data.valor), 'Tipo': data.tipo, 'Categoria': data.categoria || '', 'StatusRemocao': '', 'SolicitadoPor': '' }; sheet.appendRow(head.map(h => v[h] !== undefined ? v[h] : '')); return jsonOk({}); } function findRow(sheet, id, col) { const vals = sheet.getDataRange().getValues(); const cIdx = col ? vals[0].indexOf(col) : 0; for (let i = 1; i < vals.length; i++) { if (String(vals[i][cIdx]) === String(id)) return i + 1; } return -1; } function setStatus(ss, id, status, por) { const sheet = getGastosSheet(ss); ensureGastosHeader(sheet); const row = findRow(sheet, id, 'ID'); if (row < 0) return jsonErr('ID não encontrado'); const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const cS = head.indexOf('StatusRemocao') + 1; const cP = head.indexOf('SolicitadoPor') + 1; if (cS > 0) sheet.getRange(row, cS).setValue(status); if (cP > 0) sheet.getRange(row, cP).setValue(por); return jsonOk({}); } function deleteGasto(ss, id) { const sheet = getGastosSheet(ss); const row = findRow(sheet, id, 'ID'); if (row < 0) return jsonErr('ID não encontrado'); sheet.deleteRow(row); return jsonOk({}); } function addViagem(ss, data) { const sheet = getViagensSheet(ss); ensureViagensHeader(sheet); sheet.appendRow([ data.id, data.nome, data.inicio ? new Date(data.inicio + 'T12:00:00') : '', data.fim ? new Date(data.fim + 'T12:00:00') : '', Number(data.orcamentoDebito) || 0, Number(data.orcamentoCredito) || 0, new Date(), false ]); return jsonOk({}); } function updateViagem(ss, data) { const sheet = getViagensSheet(ss); ensureViagensHeader(sheet); const row = findRow(sheet, data.id, 'ID'); if (row < 0) return jsonErr('Viagem não encontrada'); const head = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const map = { 'Nome': data.nome, 'Inicio': data.inicio ? new Date(data.inicio + 'T12:00:00') : '', 'Fim': data.fim ? new Date(data.fim + 'T12:00:00') : '', 'OrcamentoDebito': Number(data.orcamentoDebito) || 0, 'OrcamentoCredito': Number(data.orcamentoCredito) || 0, 'Arquivada': !!data.arquivada }; Object.keys(map).forEach(k => { if (map[k] === undefined) return; const col = head.indexOf(k) + 1; if (col > 0) sheet.getRange(row, col).setValue(map[k]); }); return jsonOk({}); } function deleteViagem(ss, id) { const sheet = getViagensSheet(ss); const row = findRow(sheet, id, 'ID'); if (row < 0) return jsonErr('Viagem não encontrada'); sheet.deleteRow(row); return jsonOk({}); } function jsonOk(extra) { return ContentService.createTextOutput(JSON.stringify(Object.assign({ok:true}, extra))) .setMimeType(ContentService.MimeType.JSON); } function jsonErr(err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:String(err)})) .setMimeType(ContentService.MimeType.JSON); } // ========================================================================== // FUNÇÃO DE REPARO — executar UMA VEZ direto no editor do Apps Script: // 1) Selecione "repararPlanilha" no dropdown // 2) Clique em ▶ Executar // 3) Autorize quando pedir // O que faz: // - Remove colunas "Scope" duplicadas no cabeçalho // - Atribui ViagemId e Scope=viagem aos gastos órfãos (sem ViagemId) // usando a primeira viagem ativa da aba Viagens // - Remove linhas de teste de conexão (__teste__) // ========================================================================== function repararPlanilha() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var gSheet = getGastosSheet(ss); var vSheet = getViagensSheet(ss); var log = []; // 1) Remove colunas duplicadas no header var lastCol = gSheet.getLastColumn(); var head = gSheet.getRange(1, 1, 1, lastCol).getValues()[0]; var seen = {}; var colsToDelete = []; for (var i = 0; i < head.length; i++) { var name = String(head[i] || '').trim(); if (!name) continue; if (seen[name] !== undefined) { colsToDelete.push(i + 1); // 1-indexed } else { seen[name] = i + 1; } } // Apaga de trás pra frente pra não bagunçar índices colsToDelete.sort(function(a,b){return b-a;}).forEach(function(c){ gSheet.deleteColumn(c); log.push('Coluna duplicada removida: posição ' + c); }); // 2) Remove linhas de teste var vals = gSheet.getDataRange().getValues(); head = vals[0]; var cId = head.indexOf('ID'); var cItem = head.indexOf('Item'); var cLocal = head.indexOf('Local'); var rowsToDelete = []; for (var r = 1; r < vals.length; r++) { var id = String(vals[r][cId] || ''); var item = String(vals[r][cItem] || ''); var local = String(vals[r][cLocal] || ''); if (id.indexOf('teste-') === 0 || item.indexOf('__teste') === 0 || local === '__teste__') { rowsToDelete.push(r + 1); } } rowsToDelete.sort(function(a,b){return b-a;}).forEach(function(rr){ gSheet.deleteRow(rr); log.push('Linha de teste removida: ' + rr); }); // 3) Atribui ViagemId aos órfãos var viagens = vSheet.getDataRange().getValues(); var vHead = viagens[0] || []; var vIdCol = vHead.indexOf('ID'); var vArqCol = vHead.indexOf('Arquivada'); var viagemAtivaId = null; for (var vi = 1; vi < viagens.length; vi++) { var arq = vArqCol >= 0 ? viagens[vi][vArqCol] : false; if (!arq && viagens[vi][vIdCol]) { viagemAtivaId = String(viagens[vi][vIdCol]); break; } } if (!viagemAtivaId) { log.push('Nenhuma viagem ativa encontrada para atribuir órfãos.'); } else { log.push('Viagem ativa: ' + viagemAtivaId); // Recarrega dados após deletar linhas vals = gSheet.getDataRange().getValues(); head = vals[0]; var cV = head.indexOf('ViagemId') + 1; var cS = head.indexOf('Scope') + 1; if (cV === 0) { gSheet.getRange(1, gSheet.getLastColumn() + 1).setValue('ViagemId'); cV = gSheet.getLastColumn(); } if (cS === 0) { gSheet.getRange(1, gSheet.getLastColumn() + 1).setValue('Scope'); cS = gSheet.getLastColumn(); } var cVidx = cV - 1; var cSidx = cS - 1; var migrados = 0; for (var rr = 1; rr < vals.length; rr++) { var vid = String(vals[rr][cVidx] || '').trim(); var sc = String(vals[rr][cSidx] || '').trim().toLowerCase(); if (!vid && sc !== 'pessoal') { gSheet.getRange(rr + 1, cV).setValue(viagemAtivaId); gSheet.getRange(rr + 1, cS).setValue('viagem'); migrados++; } } log.push('Gastos migrados para a viagem ativa: ' + migrados); } // Mostra resultado num diálogo var msg = 'REPARO CONCLUÍDO\n\n' + log.join('\n'); Logger.log(msg); try { SpreadsheetApp.getUi().alert(msg); } catch (e) { /* sem UI quando rodado fora */ } }
